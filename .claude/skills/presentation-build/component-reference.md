Quick reference for implementing presentation components in Astro/HTML.

## Slide Types

### slide-title

```html
<section class="slide slide-title" id="slide-1" data-title="Introduction">
  <div class="slide-content">
    <p class="class-number">Class {N}</p>
    <h1>{Main Title}</h1>
    <p class="subtitle">{Subtitle}</p>
  </div>
</section>
```

### slide-content

```html
<section class="slide slide-content" id="slide-{n}" data-title="{Title}">
  <div class="slide-content">
    <h2>{Heading}</h2>
    <ul>
      <li>{Point 1}</li>
      <li>{Point 2}</li>
      <li>{Point 3}</li>
    </ul>
  </div>
</section>
```

### slide-visual


## JavaScript Patterns

### Basic Presentation Navigation

```javascript
(function() {
  const slides = document.querySelectorAll('.slide');
  let currentSlide = 0;

  function goToSlide(n) {
    if (n < 0 || n >= slides.length) return;

    slides[currentSlide].classList.remove('active');
    currentSlide = n;
    slides[currentSlide].classList.add('active');

    // Update progress bar
    const progress = ((currentSlide + 1) / slides.length) * 100;
    document.getElementById('progress').style.width = `${progress}%`;

    // Update hash
    window.location.hash = `slide-${currentSlide + 1}`;
  }

  function nextSlide() {
    goToSlide(currentSlide + 1);
  }

  function prevSlide() {
    goToSlide(currentSlide - 1);
  }

  function toggleTOC() {
    document.getElementById('toc').classList.toggle('active');
  }

  document.addEventListener('keydown', (e) => {
    switch(e.key) {
      case 'ArrowRight':
      case ' ':
        e.preventDefault();
        nextSlide();
        break;
      case 'ArrowLeft':
        e.preventDefault();
        prevSlide();
        break;
      case 't':
      case 'T':
        toggleTOC();
        break;
      case 'f':
      case 'F':
        if (!document.fullscreenElement) {
          document.documentElement.requestFullscreen();
        } else {
          document.exitFullscreen();
        }
        break;
      case 'Escape':
        document.getElementById('toc').classList.remove('active');
        break;
    }

    // Number keys
    if (e.key >= '0' && e.key <= '9') {
      goToSlide(parseInt(e.key));
    }
  });

  // Initialize
  goToSlide(0);

  // Expose globally
  window.Presentation = { goToSlide, nextSlide, prevSlide };
})();
```

### TOC Generation

```javascript
function generateTOC() {
  const slides = document.querySelectorAll('.slide[data-title]');
  const tocList = document.getElementById('toc-list');

  slides.forEach((slide, i) => {
    const li = document.createElement('li');
    const a = document.createElement('a');
    a.href = `#slide-${i + 1}`;
    a.textContent = `${i + 1}. ${slide.dataset.title}`;
    a.onclick = (e) => {
      e.preventDefault();
      window.Presentation.goToSlide(i);
      document.getElementById('toc').classList.remove('active');
    };
    li.appendChild(a);
    tocList.appendChild(li);
  });
}

document.addEventListener('DOMContentLoaded', generateTOC);
```

### Visualization Hook

```javascript
// Override goToSlide to trigger visualizations
const originalGoToSlide = window.Presentation.goToSlide;
window.Presentation.goToSlide = function(n) {
  originalGoToSlide(n);

  // Trigger visualizations based on slide number
  switch(n) {
    case 5:  // Funnel slide
      setTimeout(drawFunnel, 300);
      break;
    case 10: // Chart slide
      setTimeout(drawChart, 300);
      break;
  }
};
```

## Font Size Quick Reference

| Element | Minimum | Preferred |
|---------|---------|-----------|
| Slide title | 2.5rem | 3rem |
| Section heading | 2rem | 2.5rem |
| Body text | 1.5rem | 1.5rem |
| Bullet points | 1.5rem | 1.5rem |
| D3 labels | 1.25rem | 1.25rem |
| Captions | 1rem | 1.25rem |

**Never use font sizes below 1rem (16px)**
